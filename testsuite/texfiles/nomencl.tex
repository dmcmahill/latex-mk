Here is an example equation with nomenclatures:
\begin{equation}
  E = m\:.\:c^2
\end{equation}
\nomenclature{$E$}{Energy}%
\nomenclature{$m$}{Mass}%
\nomenclature{$c$}{Speed of light}%
